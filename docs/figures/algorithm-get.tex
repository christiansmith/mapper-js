% GET — the value pipeline (SPEC.md §5.5), stage-level view.
% Full per-stage semantics are in the normative text; keyword-to-stage
% assignment is illustrated in pipeline-get.svg.
\documentclass[varwidth=150mm,border=6pt]{standalone}
\usepackage[T1]{fontenc}
\usepackage{algpseudocode}
\usepackage{amsmath}
\begin{document}
\begin{algorithmic}[1]
\Function{Get}{$d$, $ctx$}
  \State $v \gets \textsc{Locate}(d, ctx)$ \Comment{pointer / scope reads; \textsc{first}/\textsc{last}/\textsc{all}; else scope pass-through}
  \State $v \gets \textsc{Dispatch}(d, v, ctx)$ \Comment{\textsc{switch}: read branch key from named scope, evaluate case}
  \State $v \gets \textsc{Plugins}(d, v, ctx)$ \Comment{\textbf{await} each registered plugin key, in document order}
  \State $v \gets \textsc{Shape}(d, v, ctx)$ \Comment{select, initialize, and transform}
  \State $\textsc{Validate}(d, v, ctx.\mathit{errors})$ \Comment{errors accumulate; $v$ passes through}
  \State $v \gets \textsc{Finalize}(d, v)$ \Comment{\textsc{default}, then coercion}
  \State \Return $v$
\EndFunction
\end{algorithmic}
\end{document}
