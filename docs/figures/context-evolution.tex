% Context derivation as data (SPEC.md §5.2, §8.2).
% Three contexts during `each` over /items with input { items: [{n:1},{n:2}] }
% (example 6.2-2). Shared fields hold the SAME objects at every level (marked
% with matching symbols, gray); rebound fields (black) are what SHIFT/NEST
% change.
\documentclass[border=8pt]{standalone}
\usepackage[T1]{fontenc}
\usepackage{amssymb}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta}
\newcommand{\shr}[1]{\textcolor{black!45}{#1}}
\begin{document}
\begin{tikzpicture}[
  ctx/.style={draw=black!60, rounded corners=3pt, inner sep=7pt, align=left,
              font=\ttfamily\footnotesize, fill=black!3},
  hdr/.style={font=\sffamily\bfseries\small},
  derive/.style={-{Stealth[length=2.8mm]}, thick, black!70},
  note/.style={font=\sffamily\footnotesize, text=black!60},
  arrlbl/.style={font=\sffamily\scriptsize, text=black!60, align=center}
]
  % three records, top-aligned
  \node[ctx] (c0) at (0,0) {%
\shr{input:\ \ \ $\bullet$}\\
\shr{output:\ \ $\blacksquare$}\\
\shr{errors:\ \ $\blacktriangle$}\\
source:\ \ $\bullet$\ \shr{(= input)}\\
target:\ \ $\blacksquare$\ \shr{(= output)}\\
paths:\ \ \ /\ $\rightarrow$\ /};

  \node[ctx, anchor=north west] (c1) at ([xshift=18mm]c0.north east) {%
\shr{input:\ \ \ $\bullet$}\\
\shr{output:\ \ $\blacksquare$}\\
\shr{errors:\ \ $\blacktriangle$}\\
source:\ \ [\{n:1\}, \{n:2\}]\\
target:\ \ $\blacksquare$\ \shr{(= output)}\\
paths:\ \ \ /items\ $\rightarrow$\ /};

  \node[ctx, anchor=north west] (c2) at ([xshift=18mm]c1.north east) {%
\shr{input:\ \ \ $\bullet$}\\
\shr{output:\ \ $\blacksquare$}\\
\shr{errors:\ \ $\blacktriangle$}\\
source:\ \ \{n:1\}\\
target:\ \ \{\}\ fresh\\
paths:\ \ \ /items/0\ $\rightarrow$\ /};

  % headers, centered above each record
  \node[hdr, above=1.5mm of c0.north] {context$_0$ · top};
  \node[hdr, above=1.5mm of c1.north] {context$_1$ · /out pairing};
  \node[hdr, above=1.5mm of c2.north] {context$_2$ · element 0};

  % title line, left-aligned with the first record, clear of the headers
  \node[note, anchor=south west] at ([yshift=9mm]c0.north west)
    {input = \texttt{\{ items: [\{n:1\}, \{n:2\}] \}}\quad(example 6.2-2)};

  % derivation arrows at a shared height, two-line centered labels
  \draw[derive] (c0.east) -- (c1.west |- c0.east)
    node[midway, above=1mm, arrlbl] {\textsc{shift}}
    node[midway, below=1mm, arrlbl] {rebind\\source};
  \draw[derive] (c1.east |- c0.east) -- (c2.west |- c0.east)
    node[midway, above=1mm, arrlbl] {\textsc{nest}}
    node[midway, below=1mm, arrlbl] {fresh target\\+ index};

  % legend, left-aligned with the first record
  \node[note, anchor=north west] at ([yshift=-4mm]c0.south west)
    {$\bullet$\ $\blacksquare$\ $\blacktriangle$\ = the \emph{same} shared objects in every context; black fields are rebound};
\end{tikzpicture}
\end{document}
